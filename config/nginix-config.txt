nano /etc/nginx/sites-available/synalyzer-console






server {
    listen 80;
    server_name 47.131.152.136;

    # PHP API — matches the hyphen URL the client uses
    location ~ ^/synalyzer-console/(.+\.php)$ {
        fastcgi_pass unix:/var/run/php/php8.2-fpm.sock;  # ← fix version here
        include fastcgi_params;
        fastcgi_param SCRIPT_FILENAME /var/www/html/synalyzer-console/$1;
        fastcgi_param SCRIPT_NAME /synalyzer-console/$1;
    }

    # Next.js Dashboard
    location / {
        proxy_pass http://localhost:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
}
